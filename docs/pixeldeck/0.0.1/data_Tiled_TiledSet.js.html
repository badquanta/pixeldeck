  static get relationMappings () {
    return {
      tileMapUsage: {
        relation: data.Model.HasManyRelation,
        modelClass: 'TileMapUsesTileSet',
        join: {
          from: 'TileSet.uuid',
          to: 'TileMapUsesTileSet.TileSetUuid'
        }
      },
      usedBy: {
        relation: data.Model.ManyToManyRelation,
        modelClass: 'TileMap',
        join: {
          from: 'TileSet.uuid',
          through: {
            className: 'TileMapUsesTileSet',
            from: 'TileMapUsesTileSet.TileSetUuid',
            to: 'TileMapUsesTileSet.TileMapUuid'
          },
          to: 'TileMap.uuid'
        }
      }
    }
