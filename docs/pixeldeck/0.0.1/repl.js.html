/**
 * @desc Initialize the Read, Evaluate, Print Loop.
 * @desc This will make the pixeldeck library instances
 * @desc available for inspection and control.
 * @note This is not an async function; but it does create a promise. It saves that promise to repl.Loading
 * @property {replClose} repl.close
 * @param {*} opts
 * @param  {...any} args
 * @emits <evt>REPL:starting
 * @emits <evt>REPL:started
 * @emits <evt>REPL:closed
 * @emits <evt>REPL:line
 * @returns {repl.REPLServer}
 */
function repl (opts = {}, ...args) {
  if (!cfg.interactive) {
    dbg('INTERACTIVE REPL DISABLED')
    return
  }
  if (repl.server && !repl.closing) {
    dbg('repl server already started')
    return repl.server
  }
  const finalReplOpts = Object.assign({}, opts, cfg.replOpts)
  dbg('starting repl(%o , %O)', opts, ...args)
  repl.server = Repl.start(
    finalReplOpts, ...args
  )
  repl.server.once('close', function () {
    dbg('closing....')
    repl.closing = true;
    require('./evt/Stop')()
  });
}

/**
 * a function that gets called to suspend repl input.
 */
function suspendRepl () {
  const alreadyTimedout = timeoutId != null
  if (timeoutId) {
    //dbg('already paused')
    clearTimeout(timeoutId)
  }
  timeoutId = setTimeout(restoreRepl, suspendTimeout)

  //dbg('pausing')
 // process.stdout.write('\n')
  process.stdin.pause()
  if (repl.server) {
    repl.server.pause()
    //repl.server.clearLine()
    //repl.server.clearBufferedCommand()
  }

}

/**
 * this is a callback function used to display the prompt again after other output.
 */
function restoreRepl () {
  timeoutId = null;
  process.stdin.resume()
  //dbg('resuming')
  if (repl.server && repl.server.paused) {
    repl.server.prompt()
  }
}
